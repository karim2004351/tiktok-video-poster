document.getElementById('postBtn').addEventListener('click', () => { const link = document.getElementById('videoLink').value.trim(); const results = document.getElementById('results'); results.innerHTML = ''; if (!link) { alert('الرجاء إدخال رابط فيديو صالح.'); return; } fetch('sites.json') .then(response => response.json()) .then(sites => { sites.forEach(site => { const div = document.createElement('div'); div.className = 'site'; div.textContent = `تم نشر الرابط على: ${site}`; results.appendChild(div); }); }) .catch(() => { alert('فشل تحميل قائمة المواقع.'); }); });
